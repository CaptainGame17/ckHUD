// ===========================================
// overheal

event HudHealthBonusPulse
{
	Animate	PlayerStatusHealthBonusImage 	Alpha		"200"					Linear 0.2 0.4
	Animate	PlayerStatusHealthBonusImage 	Alpha		"200"					Linear 0.2 0.4
	
	Animate	PlayerStatusHealthValue			FgColor		"overheal" 				Linear 0 0.1

	Animate	SpecGuiHP						FgColor		"overheal" 				Linear 0 0.1

	RunEvent HudHealthBonusPulseLoop	0.4
}

// call to loop HudHealthBonusPulse

event HudHealthBonusPulseLoop
{
	RunEvent HudHealthBonusPulse 0.0
}

event HudHealthBonusPulseStop
{
	StopEvent HudHealthBonusPulse 0.0
	StopEvent HudHealthBonusPulseLoop 0.0
	
	Animate	PlayerStatusHealthValue 		FgColor		"colorwhite"			Linear 0 0.1
	
	Animate	SpecGuiHP 						FgColor		"colorwhite"			Linear 0 0.1

}

//Health Dying Pulse

event HudHealthDyingPulse
{
	Animate	PlayerStatusHealthBonusImage 			Alpha		"150"			Linear 0 0
	
	Animate	PlayerStatusHealthValueShadow			FgColor		"low"			Linear 0.0 0.075
	Animate	PlayerStatusHealthValueShadow			FgColor		"colorblack"	Linear 0.125 0.075

	Animate	SpecGuiHPShadow							FgColor		"lowdark"		Linear 0.0 0.075
	Animate	SpecGuiHPShadow							FgColor		"colorblack"	Linear 0.125 0.075	
	
	
	RunEvent HudHealthDyingPulseLoop	0.25
}

// call to loop HudHealthDyingPulse
event HudHealthDyingPulseLoop
{
	RunEvent HudHealthDyingPulse 0.0

}

event HudHealthDyingPulseStop
{
	StopEvent HudHealthDyingPulseLoop 0.0
	StopEvent HudHealthDyingPulse 0.0

	Animate	PlayerStatusHealthValueShadow			FgColor			"colorblack" 		Linear 0.0 0.0
	Animate	SpecGuiHPShadow							FgColor			"colorblack"		Linear 0 0

}

// low ammo
event HudLowAmmoPulse
{
	Animate AmmoInClipShadow			FgColor		"low"	Linear 0.0 0.075
	Animate AmmoInReserveShadow			FgColor		"low"	Linear 0.0 0.075
	Animate AmmoNoClipShadow			FgColor		"low"	Linear 0.0 0.075
	
	Animate AmmoInClipShadow			FgColor		"colorblack"	Linear 0.125 0.075
	Animate AmmoInReserveShadow			FgColor		"colorblack"	Linear 0.125 0.075
	Animate AmmoNoClipShadow			FgColor		"colorblack"	Linear 0.125 0.075
	RunEvent HudLowAmmoPulseLoop	0.25
}

// call to loop HudLowAmmoPulse

event HudLowAmmoPulseLoop
{
	RunEvent HudLowAmmoPulse 0.0
}

event HudLowAmmoPulseStop
{

	StopEvent HudLowAmmoPulse 0.0
	StopEvent HudLowAmmoPulseLoop 0.0
	Animate AmmoInClipShadow			FgColor		"colorblack"	Linear 0 0
	Animate AmmoInReserveShadow			FgColor		"colorblack"	Linear 0 0
	Animate AmmoNoClipShadow			FgColor		"colorblack"	Linear 0 0
}
