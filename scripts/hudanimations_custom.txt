// ubercharge

event HudMedicCharged
{
	Animate	ChargeLabel 	FgColor		"colordullwhite"		spline 0.0 0.3
	Animate	ChargeLabel 	FgColor		"colorwhite"			spline 0.3 0.3
	
	Animate	ChargeMeter 	FgColor		"ubercharge1"			spline 0.0 0.3
	Animate	ChargeMeter 	FgColor		"ubercharge2"			spline 0.3 0.3
	
	Animate	ChargeColor_25 	BgColor		"Blank"					spline 0 0
	Animate	ChargeColor_50 	BgColor		"Blank"					spline 0 0
	Animate	ChargeColor_75 	BgColor		"Blank"					spline 0 0
	Animate	ChargeColor_100 BgColor		"Blank"					spline 0 0

	RunEvent HudMedicChargedLoop	0.6
}

// call to loop HudHealthBonusPulse
event HudMedicChargedLoop
{
	RunEvent HudMedicCharged 0.0
}

event HudMedicChargedStop
{
	StopEvent HudMedicCharged 0.0
	StopEvent HudMedicChargedLoop 0.0
	
	Animate	ChargeLabel 	FgColor		"colorwhite"			Linear 0 0
	Animate	ChargeMeter 	FgColor		"colorwhite"			Linear 0 0
	Animate	ChargeColor_25 	BgColor		"meter1"				Linear 0 0
	Animate	ChargeColor_50 	BgColor		"meter2"				Linear 0 0
	Animate	ChargeColor_75 	BgColor		"meter3"				Linear 0 0
	Animate	ChargeColor_100 BgColor		"meter4"				Linear 0 0
}

// overheal

event HudHealthBonusPulse
{
	Animate	PlayerStatusHealthBonusImage 	Alpha		"200"					Linear 0.2 0.4
	Animate	PlayerStatusHealthBonusImage 	Alpha		"200"					Linear 0.2 0.4
	
	Animate	PlayerStatusHealthValue			FgColor		"overheal" 				Linear 0 0

	Animate	TargetHealthBG					BgColor		"overheal" 				Linear 0 0

	RunEvent HudHealthBonusPulseLoop	0.4
}

// call to loop HudHealthBonusPulse

event HudHealthBonusPulseLoop
{
	RunEvent HudHealthBonusPulse 0.0
}

event HudHealthBonusPulseStop
{
	StopEvent HudHealthBonusPulse 0.0
	StopEvent HudHealthBonusPulseLoop 0.0
	
	Animate	PlayerStatusHealthValue 		FgColor		"colorwhite"			Linear 0 0
	
	Animate	TargetHealthBG					BgColor		"colortransparentgray" 	Linear 0 0

}

event HudHealthDyingPulse
{
	Animate	PlayerStatusHealthBonusImage 			Alpha		"150"			Linear 0 0
	
	Animate	PlayerStatusHealthValueShadow			FgColor		"low"			Linear 0.0 0.075
	Animate	PlayerStatusHealthValueShadow			FgColor		"colorblack"	Linear 0.125 0.075

	Animate	SpecGuiHPShadow			FgColor		"lowdark"		Linear 0.0 0.075
	Animate	SpecGuiHPShadow			FgColor		"colorblack"	Linear 0.125 0.075	
	
	Animate	TargetHealthBG							BgColor		"low" 			Linear 0 0
	
	RunEvent HudHealthDyingPulseLoop	0.25
}

// call to loop HudHealthDyingPulse
event HudHealthDyingPulseLoop
{
	RunEvent HudHealthDyingPulse 0.0

}

event HudHealthDyingPulseStop
{
	StopEvent HudHealthDyingPulseLoop 0.0
	StopEvent HudHealthDyingPulse 0.0

	Animate	PlayerStatusHealthValueShadow			FgColor			"colorblack" 		Linear 0.0 0.0
	Animate	SpecGuiHPShadow			FgColor			"colorblack"		Linear 0 0

	Animate	TargetHealthBG							BgColor			"colortransparentgray"	 			Linear 0 0
}

// Spy Disguise

event HudSpyDisguiseChanged
{
	Animate PlayerStatusSpyOutlineImage		Alpha		"255"			Linear 0.0 0.2
	
	Animate PlayerStatusSpyOutlineImage		Position	"c-227 c100"	Linear 0.0 0.0
	Animate PlayerStatusSpyOutlineImage		Size		"50 50"		Linear 0.0 0.0

	RunEvent HudSpyDisguiseHide	0.7
}

event HudSpyDisguiseHide
{
	
	Animate PlayerStatusSpyOutlineImage		Position	"c-227 c100"	Linear 0.0 0.0
	Animate PlayerStatusSpyOutlineImage		Size		"50 50"			Linear 0.0 0.0

	
	Animate PlayerStatusSpyOutlineImage		Alpha		"0"				Linear 0.2 0.1
}

event HudSpyDisguiseFadeIn
{
	RunEvent HudSpyDisguiseChanged	0
	Animate PlayerStatusSpyImage			Alpha		"255"			Linear 0.9 0.1	
}

event HudSpyDisguiseFadeOut
{
	RunEvent HudSpyDisguiseChanged	0
	Animate PlayerStatusSpyImage			Alpha		"0"				Linear 0.9 0.1	
}
